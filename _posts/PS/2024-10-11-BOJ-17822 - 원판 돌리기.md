> 골드 II<br>

## 문제
https://www.acmicpc.net/problem/17822

## 해결
원판 규칙을 구현합시다. => `시뮬레이션`<br>
- 원판의 회전은 원판의 시작점이 바뀌는 것과 같습니다. 원판의 시작점을 정의하고 인덱스를 참조할 때 이 시작점을 참조한 연산을 합니다. 이렇게 하면 원판 회전 연산이 O(1)이 됩니다.
- 회전 연산과 인덱스 참조 연산에는 `모듈러 연산`을 적용합니다.
- 평균 업데이트 연산은 지워진 칸 제외하고 모든 값 더해서 평균 구하고 ±1 적용하면 됩니다.

```
값 지우기
1. 원판의 모든 값 확인
	- 지워진 칸은 연산에서 고려하지 않음
	- 현재 칸과 인접한 칸 찾기
	- 하나라도 일치하면 현재 칸에 마킹
2. 마킹된 칸을 지워진 칸으로 업데이트
```

## 코드
```
#include <iostream>
using namespace std;
int N, M, T;
int start[52], board[52][52], dx[4] = { 1, 0, -1, 0 }, dm[4] = { 0, 1, 0, -1 };
bool eraseMark[52][52], eraseMarkTemp[52][52];
int mToIdx(int x, int m) { return (m - start[x] + M) % M; }

void rotate(int x, int k) {
    start[x] = (start[x] + k + M) % M;
}

void update() {
    double cnt = 0;
    double average = 0;
    for (int x = 0; x < N; x++)
        for (int m = 0; m < M; m++)
            if (!eraseMark[x][m])
                cnt++, average += board[x][m];
    if (cnt == 0)
        return;
    average /= cnt;
    for (int x = 0; x < N; x++)
        for (int m = 0; m < M; m++)
            if (!eraseMark[x][m])
                board[x][m] += (board[x][m] < average ? 1 : (board[x][m] > average ? -1 : 0));<br>
}

void erase() {
    bool isErased = 0;
    for (int x = 0; x < N; x++) {
        for (int m = 0; m < M; m++) {
            if (eraseMark[x][mToIdx(x, m)])
                continue;
            for (int d = 0; d < 4; d++) {
                if ((d == 2 && x == 0) || (d == 0 && x == N - 1))
                    continue;
                int nx = x + dx[d], nm = (m + dm[d]) % M;
                if (eraseMark[nx][mToIdx(nx, nm)])
                    continue;
                eraseMarkTemp[x][mToIdx(x, m)] |= board[x][mToIdx(x, m)] == board[nx][mToIdx(nx, nm)];
            }
            isErased |= eraseMark[x][mToIdx(x, m)] != eraseMarkTemp[x][mToIdx(x, m)];
        }
    }
    for (int x = 0; x < N; x++)
        for (int m = 0; m < M; m++)
            eraseMark[x][m] = eraseMarkTemp[x][m];

    if (!isErased)
        update();
}

int sum() {
    int sum = 0;
    for (int x = 0; x < N; x++)
        for (int m = 0; m < M; m++)
            if (!eraseMark[x][m])
                sum += board[x][m];
    return sum;
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    cin >> N >> M >> T;<br>
    for (int x = 0; x < N; x++)
        for (int m = 0; m < M; m++)
            cin >> board[x][m];<br>
    while (T--) {
        int x, d, k;
        cin >> x >> d >> k;<br>
        for (int i = 1; x * i <= N; i++)
            rotate(x * i - 1, k * (d == 0 ? 1 : -1));
        erase();
    }
    cout << sum();
    return 0;
}
```
