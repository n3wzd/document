> 브론즈 III<br>

## 문제
https://www.acmicpc.net/problem/10810<br>

## 코드
```
#include <iostream>
using namespace std;
int N, M, basket[102];

int main() {
	cin >> N >> M;<br>
	while (M--) {
		int i, j, k;
		cin >> i >> j >> k;<br>
		for (int x = i; x <= j; x++)
			basket[x] = k;
	}
	for (int x = 1; x <= N; x++)
		cout << basket[x] << " ";
	return 0;
}
```
