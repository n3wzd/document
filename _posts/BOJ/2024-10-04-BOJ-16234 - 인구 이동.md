> 골드 IV<br>

## 문제
https://www.acmicpc.net/problem/16234<br>

## 해결
조건에 맞는 인접한 칸이 그룹을 형성해야 합니다. `그래프 탐색`을 사용합시다. (DFS, BFS 상관 없음)

```
1. 모든 칸을 검사합니다.
	- 현재 칸에 대해 DFS 탐색을 시작합니다.
	- 조건(L 이상, R 이하 차이)에 맞는 정점들을 모두 찾습니다.
	- 정점을 모두 찾았다면 연합 인구수를 계산하고 대입합니다.
		- 대입할 때, 다시 한번 DFS를 활용합니다.
2. 연합 크기가 2 이상인 것이 하나라도 있다면 탐색 반복
	- 그렇지 않으면 탐색 종료
```

## 코드
```
#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;
int N, L, R, popular, cnt, res;
int board[52][52], visited[52][52];
int dx[4] = { 1, 0, -1, 0 }, dy[4] = { 0, 1, 0, -1, };

void DFS(int x, int y) {
	visited[y][x] = 1;
	popular += board[y][x]; cnt++;
	for (int d = 0; d < 4; d++) {
		int nx = x + dx[d], ny = y + dy[d];
		if (nx < 0 || nx >= N || ny < 0 || ny >= N || visited[ny][nx] != 0)
			continue;
		int diff = abs(board[y][x] - board[ny][nx]);
		if (diff >= L && diff <= R)
			DFS(nx, ny);
	}
}

void DFS2(int x, int y, int value) {
	visited[y][x] = 2;
	board[y][x] = value;
	for (int d = 0; d < 4; d++) {
		int nx = x + dx[d], ny = y + dy[d];
		if (nx < 0 || nx >= N || ny < 0 || ny >= N || visited[ny][nx] != 1)
			continue;
		DFS2(nx, ny, value);
	}
}

int main() {
	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
	cin >> N >> L >> R;<br>
	for (int y = 0; y < N; y++)
		for (int x = 0; x < N; x++)
			cin >> board[y][x];<br>
	while (1) {
		memset(visited, 0, sizeof(visited));
		bool hasNone = 1;
		for (int y = 0; y < N; y++) {
			for (int x = 0; x < N; x++) {
				if (visited[y][x] == 0) {
					popular = 0, cnt = 0;
					DFS(x, y);
					DFS2(x, y, popular / cnt);
					hasNone &= cnt == 1;
				}
			}
		}
		if (hasNone)
			break;
		res++;
	}
	cout << res;
	return 0;
}
```
