> 골드 IV<br>

## 문제
https://www.acmicpc.net/problem/17140<br>

## 해결
주어진 조건을 분석해서 그대로 구현하면 됩니다. 중간에 `정렬`이 필요합니다.

```
연산 과정 (행/열 최대 길이에 따라 R or C연산 수행)
1. 행/열을 탐색해서 등장 횟수 배열 업데이트 (인덱스: 수, 값: 등장 횟수)
2. 등장 횟수 배열을 통해 임시 배열 생성
	- 임시 배열 값: { 수, 등장 횟수 }
3. 임시 배열을 등장 횟수 오름차순, 수 오름차순으로 정렬
4. 정렬된 임시 배열을 기반으로 원래 행/열 업데이트
5. 변경된 행/열의 최대 길이 업데이트 (최대 100)
```

## 코드
```
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
struct Node { int n = 0, c = 0; };
int r, c, k, R = 3, C = 3, phase, tmpLen;
int board[102][102], cnt[102];
Node tmp[102];

void updateTmp() {
	tmpLen = 0;
	for (int i = 1; i <= 100; i++)
		if (cnt[i] > 0)<br>
			tmp[tmpLen++] = { i, cnt[i] };
	sort(tmp, tmp + tmpLen, [](const Node& a, const Node& b) { return a.c != b.c ? a.c < b.c : a.n < b.n; });
}

int main() {
	ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
	cin >> r >> c >> k;<br>
	for (int y = 0; y < R; y++)
		for (int x = 0; x < C; x++)
			cin >> board[y][x];<br>
	while (phase <= 100) {
		if (board[r - 1][c - 1] == k) {
			cout << phase;
			return 0;
		}
		if (R >= C) {
			for (int y = 0; y < R; y++) {
				memset(cnt, 0, sizeof(cnt));
				for (int x = 0; x < C; x++)
					cnt[board[y][x]]++;
				updateTmp();
				for (int i = 0; i < 50; i++) {
					if (i < tmpLen) {
						board[y][i * 2] = tmp[i].n;
						board[y][i * 2 + 1] = tmp[i].c;
					}
					else
						board[y][i * 2 + 1] = board[y][i * 2] = 0;
				}
				C = min(max(C, tmpLen * 2), 100);
			}
		} else {
			for (int x = 0; x < C; x++) {
				memset(cnt, 0, sizeof(cnt));
				for (int y = 0; y < R; y++)
					cnt[board[y][x]]++;
				updateTmp();
				for (int i = 0; i < 50; i++) {
					if (i < tmpLen) {
						board[i * 2][x] = tmp[i].n;
						board[i * 2 + 1][x] = tmp[i].c;
					}
					else
						board[i * 2 + 1][x] = board[i * 2][x] = 0;
				}
				R = min(max(R, tmpLen * 2), 100);
			}
		}
		phase++;
	}
	cout << -1;
	return 0;
}
```
